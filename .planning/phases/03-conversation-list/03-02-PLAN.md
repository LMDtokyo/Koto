Create `SwipeableConversationItem.kt`. This wraps `ConversationItem` with a horizontal swipe layer that reveals action panels behind it.

### Architecture

Use `AnchoredDraggableState` (Compose Foundation — already in BOM 2026.03.01). This avoids gesture vs. scroll conflict (PITFALLS A2) because AnchoredDraggable participates in the `nestedScroll` system automatically.

The swipeable item is a `Box` with:
1. **Behind layer** (action panel): a Row showing the appropriate action based on swipe direction.
2. **Front layer** (content): `ConversationItem` offset by `draggableState.offset`.

Action widths per FEATURES.md: 72dp for first action, 144dp full reveal.

### SwipeAnchor enum


### Action panel colors (semantic — NOT hardcoded):
- Archive (left swipe): `Color(0xFF0D9488)` — teal. Not in KotoTheme semantic tokens, so define as local val within the composable context. This is a gesture-action color, not a brand token.
- Pin (right swipe first): `Color(0xFFF59E0B)` — amber.
- Mute (right swipe second, beyond 144dp): `Color(0xFF6B7280)` — gray.
- Red (delete, revealed only at full 144dp): `KotoTheme.colors.error`.

Per FEATURES.md swipe spec, right swipe reveals pin (amber bg) + mute (gray bg) sequentially as swipe extends.

### Full implementation

```kotlin
@OptIn(ExperimentalFoundationApi::class)
@Composable
fun SwipeableConversationItem(
    conv      : ConversationUi,
    onClick   : () -> Unit,
    onArchive : (String) -> Unit,
    onPin     : (String) -> Unit,
    onMute    : (String) -> Unit,
    modifier  : Modifier = Modifier,
) {
    val density = LocalDensity.current
    val view    = LocalView.current
    val colors  = KotoTheme.colors

    val actionWidthPx = with(density) { 72.dp.toPx() }

    val anchors = remember(actionWidthPx) {
        DraggableAnchors {
            SwipeAnchor.Center at 0f
            SwipeAnchor.Left   at -actionWidthPx
            SwipeAnchor.Right  at  actionWidthPx
        }
    }

    val draggableState = remember {
        AnchoredDraggableState(
            initialValue       = SwipeAnchor.Center,
            anchors            = anchors,
            positionalThreshold = { d -> d * 0.5f },
            velocityThreshold   = { with(density) { 125.dp.toPx() } },
            snapAnimationSpec  = spring(
                stiffness    = Spring.StiffnessMedium,
                dampingRatio = Spring.DampingRatioNoBouncy,
            ),
            decayAnimationSpec = exponentialDecay(),
        )
    }

    // Haptic: fire once per threshold crossing (track previous crossing)
    val hapticFired = remember { mutableStateOf(false) }
    val offset = draggableState.offset
    LaunchedEffect(offset) {
        val crossed = kotlin.math.abs(offset) >= actionWidthPx * 0.5f
        if (crossed && !hapticFired.value) {
            view.performHapticFeedback(HapticFeedbackConstants.CONTEXT_CLICK)
            hapticFired.value = true
        } else if (!crossed) {
            hapticFired.value = false
        }
    }

    // Commit haptic when anchored to Left or Right
    LaunchedEffect(draggableState.currentValue) {
        if (draggableState.currentValue != SwipeAnchor.Center) {
            val feedbackType = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R)
                HapticFeedbackConstants.CONFIRM
            else
                HapticFeedbackConstants.LONG_PRESS
            view.performHapticFeedback(feedbackType)
        }
    }

    Box(modifier = modifier.fillMaxWidth()) {
        // Behind: action panels
        SwipeActionBackground(
            offset    = offset,
            threshold = actionWidthPx,
            onArchive = { onArchive(conv.id); /* reset */ },
            onPin     = { onPin(conv.id) },
            onMute    = { onMute(conv.id) },
            errorColor = colors.error,
        )

        // Front: conversation item offset by drag
        ConversationItem(
            conv     = conv,
            onClick  = onClick,
            modifier = Modifier
                .offset { IntOffset(offset.roundToInt(), 0) }
                .anchoredDraggable(draggableState, Orientation.Horizontal),
        )
    }
}
```

`Modifier.offset { IntOffset(offset.roundToInt(), 0) }` — uses the lambda form to defer state read to layout phase (PITFALLS P3 — not `.offset(x = offset.dp)`).

### SwipeActionBackground

```kotlin
@Composable
private fun SwipeActionBackground(
    offset    : Float,
    threshold : Float,
    onArchive : () -> Unit,
    onPin     : () -> Unit,
    onMute    : () -> Unit,
    errorColor: Color,
) {
    val archiveTeal = Color(0xFF0D9488)
    val pinAmber    = Color(0xFFF59E0B)
    val muteGray    = Color(0xFF6B7280)

    Box(modifier = Modifier.matchParentSize()) {
        if (offset < 0f) {
            // Left swipe: archive on the right side
            Row(
                modifier = Modifier
                    .fillMaxHeight()
                    .width(72.dp)
                    .align(Alignment.CenterEnd)
                    .background(archiveTeal)
                    .clickable { onArchive() },
                verticalAlignment = Alignment.CenterVertically,
                horizontalArrangement = Arrangement.Center,
            ) {
                Icon(Icons.Default.Archive, contentDescription = "Archive", tint = Color.White)
            }
        } else if (offset > 0f) {
            // Right swipe: pin + mute on the left side
            Row(
                modifier = Modifier
                    .fillMaxHeight()
                    .width(144.dp)
                    .align(Alignment.CenterStart),
            ) {
                Box(
                    modifier = Modifier
                        .fillMaxHeight()
                        .weight(1f)
                        .background(pinAmber)
                        .clickable { onPin() },
                    contentAlignment = Alignment.Center,
                ) {
                    Icon(Icons.Default.PushPin, contentDescription = "Pin", tint = Color.White)
                }
                Box(
                    modifier = Modifier
                        .fillMaxHeight()
                        .weight(1f)
                        .background(muteGray)
                        .clickable { onMute() },
                    contentAlignment = Alignment.Center,
                ) {
                    Icon(Icons.Default.VolumeOff, contentDescription = "Mute", tint = Color.White)
                }
            }
        }
    }
}
```
