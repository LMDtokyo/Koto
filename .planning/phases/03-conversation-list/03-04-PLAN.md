Create `PullToRefresh.kt`. This wraps Material3's `PullToRefreshBox` with a custom `KotoBlobIndicator` that implements the liquid/spring animation described in FEATURES.md.

### Animation description (from FEATURES.md)
1. As user pulls down, a blob of Primary color stretches from the top edge.
2. At 72dp threshold, blob detaches and forms a circle. Spring bounce on detach.
3. During loading: circle pulses (scale 1.0 → 1.05, 800ms cycle).
4. On complete: circle shrinks to 0 with spring.

### Implementation strategy

Use `PullToRefreshBox` for gesture handling (it manages the pull progress float correctly). Provide a custom `indicator` composable that reads `PullToRefreshState.distanceFraction` to drive the blob morph.

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun KotoPullToRefresh(
    isRefreshing : Boolean,
    onRefresh    : () -> Unit,
    modifier     : Modifier = Modifier,
    content      : @Composable () -> Unit,
) {
    val state = rememberPullToRefreshState()

    // Haptic at threshold
    val view = LocalView.current
    var hapticFired by remember { mutableStateOf(false) }
    LaunchedEffect(state.distanceFraction) {
        val triggered = state.distanceFraction >= 1f
        if (triggered && !hapticFired) {
            val feedbackType = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R)
                HapticFeedbackConstants.CONFIRM
            else
                HapticFeedbackConstants.LONG_PRESS
            view.performHapticFeedback(feedbackType)
            hapticFired = true
        } else if (!triggered) {
            hapticFired = false
        }
    }

    PullToRefreshBox(
        isRefreshing = isRefreshing,
        onRefresh    = onRefresh,
        state        = state,
        modifier     = modifier,
        indicator    = {
            KotoBlobIndicator(
                state        = state,
                isRefreshing = isRefreshing,
                modifier     = Modifier.align(Alignment.TopCenter),
            )
        },
        content = content,
    )
}
```

### KotoBlobIndicator

The indicator uses a `Canvas` composable to draw the morphing blob.

```kotlin
@OptIn(ExperimentalMaterial3Api::class)
@Composable
private fun KotoBlobIndicator(
    state        : PullToRefreshState,
    isRefreshing : Boolean,
    modifier     : Modifier = Modifier,
) {
    val primaryColor = KotoTheme.colors.primary
    val motion       = KotoTheme.motion

    // Blob shape: morphs from stretched oval (pulling) → circle (threshold)
    // progress: 0f = no pull, 1f = threshold reached
    val progress = state.distanceFraction.coerceIn(0f, 1f)

    // Detached circle scale (pulsing during refresh)
    val pulseScale by rememberInfiniteTransition(label = "refreshPulse")
        .animateFloat(
            initialValue  = 1f,
            targetValue   = 1.05f,
            animationSpec = infiniteRepeatable(
                animation  = tween(durationMillis = 800, easing = FastOutSlowInEasing),
                repeatMode = RepeatMode.Reverse,
            ),
            label = "pulseScale",
        )

    // Shrink-to-zero when refresh completes: animate from 1f to 0f
    val completionScale by animateFloatAsState(
        targetValue   = if (isRefreshing) 1f else 0f,
        animationSpec = if (isRefreshing) spring(
            dampingRatio = Spring.DampingRatioLowBouncy,
            stiffness    = Spring.StiffnessMedium,
        ) else spring(
            dampingRatio = Spring.DampingRatioNoBouncy,
            stiffness    = Spring.StiffnessMediumLow,
        ),
        label = "completionScale",
    )

    val circleRadius = 12.dp  // 24dp diameter indicator circle

    Canvas(
        modifier = modifier
            .padding(top = 4.dp)
            .size(width = 32.dp, height = 48.dp)
    ) {
        val centerX = size.width / 2f
        val radiusPx = circleRadius.toPx()

        if (isRefreshing) {
            // Pulsing circle during loading
            val r = radiusPx * pulseScale * completionScale
            drawCircle(
                color  = primaryColor,
                radius = r,
                center = Offset(centerX, radiusPx + 4.dp.toPx()),
            )
        } else if (progress > 0f) {
            // Blob morphing: stretched oval → circle as progress → 1.0
            val circleY   = radiusPx + 4.dp.toPx()
            val blobH     = radiusPx * (1f + (1f - progress) * 1.5f)  // taller when less progress
            val blobW     = radiusPx * (0.6f + progress * 0.4f)        // narrower when less progress
            val alpha     = (progress * 2f).coerceIn(0f, 1f)           // fade in during first 50% pull

            drawOval(
                color  = primaryColor.copy(alpha = alpha),
                topLeft = Offset(centerX - blobW, circleY - blobH),
                size    = Size(blobW * 2f, blobH * 2f),
            )
        }
    }
}
```
