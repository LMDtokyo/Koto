    ```kotlin
    // ─── Data ────────────────────────────────────────────────────────────────────

    private data class OnboardingSlide(
        val lottieRes   : Int,
        val title       : String,
        val description : String,
    )

    private val SLIDES = listOf(
        OnboardingSlide(
            lottieRes   = R.raw.lottie_security,
            title       = "Надёжное шифрование",
            description = "Каждое сообщение защищено протоколом Signal. Только вы и собеседник читаете переписку.",
        ),
        OnboardingSlide(
            lottieRes   = R.raw.lottie_privacy,
            title       = "Полная анонимность",
            description = "Никакого номера телефона. Ваш адрес — публичный ключ. Сервер не знает, кто вы.",
        ),
        OnboardingSlide(
            lottieRes   = R.raw.lottie_speed,
            title       = "Молниеносная скорость",
            description = "120fps интерфейс, мгновенная доставка через WebSocket. Ощущение нативного iOS.",
        ),
    )

    // ─── IllustratedSlides ────────────────────────────────────────────────────────

    @Composable
    private fun IllustratedSlides(
        onFinish : () -> Unit,
        onSkip   : () -> Unit,
    ) {
        val pagerState = rememberPagerState(pageCount = { SLIDES.size })
        val scope      = rememberCoroutineScope()
        val isLastPage = pagerState.currentPage == SLIDES.lastIndex

        Box(modifier = Modifier.fillMaxSize()) {
            // Skip button — top-right
            TextButton(
                onClick  = onSkip,
                modifier = Modifier
                    .align(Alignment.TopEnd)
                    .padding(top = 16.dp, end = 16.dp)
                    .windowInsetsPadding(WindowInsets.statusBars),
            ) {
                Text(
                    "Пропустить",
                    style = MaterialTheme.typography.labelLarge,
                    color = KotoTheme.colors.onSurfaceLow,
                )
            }

            Column(
                modifier            = Modifier
                    .fillMaxSize()
                    .padding(horizontal = 24.dp),
                horizontalAlignment = Alignment.CenterHorizontally,
                verticalArrangement = Arrangement.Center,
            ) {
                Spacer(Modifier.height(64.dp))

                // ── Pager ────────────────────────────────────────────────────
                HorizontalPager(
                    state    = pagerState,
                    modifier = Modifier
                        .fillMaxWidth()
                        .weight(1f),
                ) { page ->
                    SlideContent(slide = SLIDES[page])
                }

                // ── Page indicator dots ──────────────────────────────────────
                PageDots(
                    count   = SLIDES.size,
                    current = pagerState.currentPage,
                )

                Spacer(Modifier.height(24.dp))

                // ── Next / Get Started button ────────────────────────────────
                Button(
                    onClick = {
                        if (isLastPage) onFinish()
                        else scope.launch {
                            pagerState.animateScrollToPage(pagerState.currentPage + 1)
                        }
                    },
                    modifier = Modifier
                        .fillMaxWidth()
                        .height(54.dp),
                    shape  = RoundedCornerShape(14.dp),
                    colors = ButtonDefaults.buttonColors(
                        containerColor = KotoTheme.colors.primary
                    ),
                ) {
                    Text(
                        text       = if (isLastPage) "Начать" else "Далее",
                        fontWeight = FontWeight.SemiBold,
                        fontSize   = 16.sp,
                        color      = Color.White,
                    )
                }

                Spacer(Modifier.height(32.dp))
            }
        }
    }

    @Composable
    private fun SlideContent(slide: OnboardingSlide) {
        val composition by rememberLottieComposition(
            LottieCompositionSpec.RawRes(slide.lottieRes)
        )
        Column(
            modifier            = Modifier.fillMaxWidth(),
            horizontalAlignment = Alignment.CenterHorizontally,
            verticalArrangement = Arrangement.spacedBy(24.dp),
        ) {
            LottieAnimation(
                composition = composition,
                iterations  = LottieConstants.IterateForever,
                modifier    = Modifier
                    .size(260.dp)
                    .padding(horizontal = 16.dp),
            )
            Column(
                horizontalAlignment = Alignment.CenterHorizontally,
                verticalArrangement = Arrangement.spacedBy(10.dp),
            ) {
                Text(
                    text      = slide.title,
                    style     = KotoTheme.typography.displayMedium,
                    color     = KotoTheme.colors.onBackground,
                    textAlign = TextAlign.Center,
                )
                Text(
                    text      = slide.description,
                    style     = KotoTheme.typography.bodyMedium,
                    color     = KotoTheme.colors.onSurfaceLow,
                    textAlign = TextAlign.Center,
                    modifier  = Modifier.padding(horizontal = 8.dp),
                )
            }
        }
    }

    @Composable
    private fun PageDots(count: Int, current: Int) {
        Row(
            horizontalArrangement = Arrangement.spacedBy(8.dp),
            verticalAlignment     = Alignment.CenterVertically,
        ) {
            repeat(count) { index ->
                val isActive = index == current
                val dotWidth by animateDpAsState(
                    targetValue = if (isActive) 24.dp else 8.dp,
                    animationSpec = spring(
                        stiffness    = 400f,
                        dampingRatio = 0.70f,
                    ),
                    label = "dot_width_$index",
                )
                Box(
                    modifier = Modifier
                        .height(8.dp)
                        .width(dotWidth)
                        .clip(RoundedCornerShape(4.dp))
                        .background(
                            if (isActive) KotoTheme.colors.primary
                            else KotoTheme.colors.onSurfaceMuted
                        )
                )
            }
        }
    }
    ```
