    ```kotlin
    /** Called after a permission dialog completes (granted or denied) or when user taps Skip. */
    fun advancePermissions(current: OnboardingStep) {
        val next = when (current) {
            OnboardingStep.PermissionNotification -> OnboardingStep.PermissionCamera
            OnboardingStep.PermissionCamera       -> OnboardingStep.PermissionMic
            OnboardingStep.PermissionMic          -> null  // done
            else                                  -> null
        }
        if (next != null) {
            _state.update { it.copy(step = next) }
        } else {
            _state.update { it.copy(registered = true) }
        }
    }
    ```

    ```kotlin
    @Composable
    fun PermissionExplanationScreen(
        config      : PermissionConfig,
        onAdvance   : () -> Unit,
    ) {
        val permissionLauncher = rememberLauncherForActivityResult(
            contract = ActivityResultContracts.RequestPermission()
        ) {
            // Advance regardless of whether user granted or denied
            onAdvance()

        var visible by remember { mutableStateOf(false) }
        LaunchedEffect(Unit) {
            kotlinx.coroutines.delay(60)
            visible = true

                    contentAlignment = Alignment.Center,
    ```
