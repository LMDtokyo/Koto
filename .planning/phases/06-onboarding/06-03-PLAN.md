    Wait — the original `startKeyGen()` is triggered by WelcomePage's "Start" button. We need a different flow:
    - WelcomePage "Начать" → calls `viewModel::startKeyGen` which now goes to `DisplayName`
    - DisplayName page "Продолжить" → calls `viewModel::confirmDisplayName` which goes to `GeneratingKey` and runs crypto

    So rename/refactor:

    ```kotlin
    /** Called from WelcomePage "Начать" — advances to display name entry. */
    fun goToDisplayName() {
        _state.update { it.copy(step = OnboardingStep.DisplayName, displayNameError = null) }
    }

    /** Called on each keystroke in the display name field. */
    fun updateDisplayName(value: String) {
        val error = validateDisplayName(value)
        _state.update { it.copy(displayName = value, displayNameError = if (value.isEmpty()) null else error) }
    }

    /** Called from DisplayNamePage "Продолжить" — validates then starts key generation. */
    fun confirmDisplayName() {
        val name  = _state.value.displayName.trim()
        val error = validateDisplayName(name)
        if (error != null) {
            _state.update { it.copy(displayNameError = error) }
            return
        }
        viewModelScope.launch {
            _state.update { it.copy(step = OnboardingStep.GeneratingKey, displayNameError = null) }
            val keys   = keyManager.generateIdentityKey()
            delay(1200)
            val result = authRepository.register(keys)
            result.onSuccess { accountId ->
                _state.update { it.copy(step = OnboardingStep.ShowAccountId, accountId = accountId) }
            }.onFailure { err ->
                _state.update { it.copy(step = OnboardingStep.DisplayName, error = err.message) }
            }
        }
    }

    private fun validateDisplayName(value: String): String? {
        val trimmed = value.trim()
        return when {
            trimmed.isEmpty()   -> "Введите имя"
            trimmed.length < 3  -> "Минимум 3 символа"
            trimmed.length > 30 -> "Максимум 30 символов"
            else                -> null
        }
    }
    ```

    Remove the old `startKeyGen()` function (it is fully replaced by `goToDisplayName()` + `confirmDisplayName()`).

    ```kotlin
    // ─── Display Name Page ─────────────────────────────────────────────────────────

    @Composable
    private fun DisplayNamePage(
        displayName      : String,
        displayNameError : String?,
        onValueChange    : (String) -> Unit,
        onContinue       : () -> Unit,
    ) {
        val isValid           = displayNameError == null && displayName.trim().length >= 3
        val focusRequester    = remember { FocusRequester() }
        val keyboardController = LocalSoftwareKeyboardController.current

        LaunchedEffect(Unit) { focusRequester.requestFocus() }

        Column(
            modifier            = Modifier
                .fillMaxWidth()
                .padding(horizontal = 32.dp),
            horizontalAlignment = Alignment.CenterHorizontally,
            verticalArrangement = Arrangement.spacedBy(0.dp),
        ) {
            Spacer(Modifier.height(56.dp))

            Text(
                text      = "Как вас зовут?",
                style     = MaterialTheme.typography.headlineMedium,
                color     = TextPrimary,
                textAlign = TextAlign.Center,
            )

            Spacer(Modifier.height(8.dp))

            Text(
                text      = "Это имя увидят ваши контакты. Его можно изменить позже.",
                style     = MaterialTheme.typography.bodyMedium,
                color     = TextSecondary,
                textAlign = TextAlign.Center,
            )

            Spacer(Modifier.height(32.dp))

            // ── Input field ────────────────────────────────────────────────────
            val borderColor by animateColorAsState(
                targetValue    = when {
                    displayNameError != null -> Color(0xFFF87171)   // KotoPalette.errorRed
                    isValid                  -> Color(0xFF34D399)   // KotoPalette.onlineGreen
                    else                     -> Color(0xFF3D3D50)   // divider
                },
                animationSpec  = tween(200),
                label          = "border_color",
            )

            OutlinedTextField(
                value         = displayName,
                onValueChange = { if (it.length <= 30) onValueChange(it) },
                modifier      = Modifier
                    .fillMaxWidth()
                    .focusRequester(focusRequester),
                placeholder   = {
                    Text("Ваше имя", color = TextTertiary)
                },
                singleLine    = true,
                keyboardOptions = KeyboardOptions(
                    capitalization = KeyboardCapitalization.Words,
                    imeAction      = ImeAction.Done,
                ),
                keyboardActions = KeyboardActions(
                    onDone = {
                        keyboardController?.hide()
                        onContinue()
                    }
                ),
                trailingIcon = {
                    AnimatedVisibility(
                        visible = isValid,
                        enter   = scaleIn(animationSpec = spring(stiffness = 400f, dampingRatio = 0.70f)),
                        exit    = scaleOut(animationSpec = tween(100)),
                    ) {
                        Icon(
                            imageVector = Icons.Default.CheckCircle,
                            contentDescription = null,
                            tint     = Color(0xFF34D399),
                            modifier = Modifier.size(20.dp),
                        )
                    }
                },
                colors = OutlinedTextFieldDefaults.colors(
                    focusedBorderColor   = borderColor,
                    unfocusedBorderColor = borderColor,
                    cursorColor          = AccentPrimary,
                    focusedTextColor     = TextPrimary,
                    unfocusedTextColor   = TextPrimary,
                    focusedContainerColor   = BgSurface1,
                    unfocusedContainerColor = BgSurface1,
                ),
                shape = RoundedCornerShape(12.dp),
            )

            // ── Animated error message ─────────────────────────────────────────
            AnimatedVisibility(
                visible = displayNameError != null,
                enter   = slideInVertically(
                    animationSpec = spring(stiffness = 400f, dampingRatio = 0.70f)
                ) { -it / 2 } + fadeIn(tween(150)),
                exit    = slideOutVertically { -it / 2 } + fadeOut(tween(100)),
            ) {
                Row(
                    modifier              = Modifier
                        .fillMaxWidth()
                        .padding(top = 6.dp, start = 4.dp),
                    horizontalArrangement = Arrangement.spacedBy(4.dp),
                    verticalAlignment     = Alignment.CenterVertically,
                ) {
                    Icon(
                        imageVector = Icons.Default.ErrorOutline,
                        contentDescription = null,
                        tint     = Color(0xFFF87171),
                        modifier = Modifier.size(14.dp),
                    )
                    Text(
                        text  = displayNameError ?: "",
                        style = MaterialTheme.typography.labelMedium,
                        color = Color(0xFFF87171),
                    )
                }
            }

            Spacer(Modifier.height(24.dp))

            // ── Character counter ────────────────────────────────────────────
            Text(
                text     = "${displayName.length}/30",
                style    = MaterialTheme.typography.labelSmall,
                color    = TextTertiary,
                modifier = Modifier
                    .align(Alignment.End)
                    .padding(end = 4.dp),
            )

            Spacer(Modifier.height(12.dp))

            // ── Continue button ──────────────────────────────────────────────
            Button(
                onClick   = {
                    keyboardController?.hide()
                    onContinue()
                },
                enabled   = displayName.trim().length >= 3,
                modifier  = Modifier.fillMaxWidth().height(54.dp),
                shape     = RoundedCornerShape(14.dp),
                colors    = ButtonDefaults.buttonColors(
                    containerColor         = AccentPrimary,
                    disabledContainerColor = AccentPrimary.copy(alpha = 0.38f),
                ),
            ) {
                Text(
                    "Продолжить",
                    fontWeight    = FontWeight.SemiBold,
                    fontSize      = 16.sp,
                    color         = Color.White,
                    letterSpacing = 0.2.sp,
                )
            }
        }
    }
    ```
