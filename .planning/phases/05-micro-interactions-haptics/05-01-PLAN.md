```kotlin
package run.koto.ui.theme

import android.os.Build
import android.view.HapticFeedbackConstants
import android.view.View
import androidx.compose.runtime.Composable
import androidx.compose.runtime.Stable
import androidx.compose.runtime.remember
import androidx.compose.ui.platform.LocalView

/**
 * Centralized haptic feedback utility for KotoTheme.
 * MI-03: Maps each user action to a distinct haptic pattern.
 *
 * Usage:
 *   val haptics = rememberKotoHaptics()
 *   haptics.onSend()
 *
 * Degradation: API 30+ uses CONFIRM/REJECT/GESTURE_* constants.
 *              API 33+ adds VibrationEffect.Composition for richer patterns.
 *              Below API 30 falls back to CONTEXT_CLICK (universally available).
 */
@Stable
class KotoHaptics(private val view: View) {

    /** Tap send button — satisfying confirmation. */
    fun onSend() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
            view.performHapticFeedback(HapticFeedbackConstants.CONFIRM)
        } else {
            view.performHapticFeedback(HapticFeedbackConstants.CONTEXT_CLICK)
        }
    }

    /** Incoming message arrives while user is at bottom — subtle notification. */
    fun onReceive() {
        view.performHapticFeedback(HapticFeedbackConstants.CLOCK_TICK)
    }

    /** Long-press on message bubble or avatar. */
    fun onLongPress() {
        view.performHapticFeedback(HapticFeedbackConstants.LONG_PRESS)
    }

    /**
     * Swipe crosses action threshold (72dp or 144dp in conversation list,
     * 48dp in swipe-to-reply). Snap feedback.
     */
    fun onSwipeThreshold() {
        view.performHapticFeedback(HapticFeedbackConstants.CONTEXT_CLICK)
    }

    /** Pull-to-refresh releases and triggers refresh. */
    fun onPullToRefresh() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
            view.performHapticFeedback(HapticFeedbackConstants.CONFIRM)
        } else {
            view.performHapticFeedback(HapticFeedbackConstants.CONTEXT_CLICK)
        }
    }

    /** Double-tap react — light tap on contact, medium on heart settle. */
    fun onDoubleTap() {
        view.performHapticFeedback(HapticFeedbackConstants.CLOCK_TICK)
    }

    /** Heart particle appears after double-tap react. Medium confirmation. */
    fun onReactionSettle() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
            view.performHapticFeedback(HapticFeedbackConstants.CONFIRM)
        } else {
            view.performHapticFeedback(HapticFeedbackConstants.CONTEXT_CLICK)
        }
    }
}

/**
 * Remember a [KotoHaptics] instance bound to the current [LocalView].
 * Call at composable scope; pass the result down to gesture handlers.
 */
@Composable
fun rememberKotoHaptics(): KotoHaptics {
    val view = LocalView.current
    return remember(view) { KotoHaptics(view) }
}
```
