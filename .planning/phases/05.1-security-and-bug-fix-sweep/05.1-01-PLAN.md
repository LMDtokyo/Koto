    ```go
    // Package middleware — rate limit tiers backed by github.com/go-chi/httprate.
    package middleware

    import (
        "encoding/json"
        "fmt"
        "net/http"
        "time"

        "github.com/go-chi/httprate"
    )

    // PublicLimiter enforces 5 req/min per real IP on unauthenticated routes
    // (/v1/auth/*) — prevents registration / token-refresh flooding.
    func PublicLimiter() func(http.Handler) http.Handler {
        return httprate.Limit(
            5,
            time.Minute,
            httprate.WithKeyFuncs(httprate.KeyByRealIP),
            httprate.WithLimitHandler(rateLimitResponse),
        )
    }

    // AuthedWriteLimiter enforces 60 req/min per account for write-side APIs
    // (conversations, contacts, bot registration, device registration).
    // Keyed on X-Account-ID injected upstream by JWTAuth.
    func AuthedWriteLimiter() func(http.Handler) http.Handler {
        return httprate.Limit(
            60,
            time.Minute,
            httprate.WithKeyFuncs(keyByAccountID),
            httprate.WithLimitHandler(rateLimitResponse),
        )
    }

    // AuthedReadLimiter enforces 300 req/min per account for read-side APIs
    // (user profile lookup, pre-key bundle fetch, media metadata).
    func AuthedReadLimiter() func(http.Handler) http.Handler {
        return httprate.Limit(
            300,
            time.Minute,
            httprate.WithKeyFuncs(keyByAccountID),
            httprate.WithLimitHandler(rateLimitResponse),
        )
    }

    func keyByAccountID(r *http.Request) (string, error) {
        id := r.Header.Get("X-Account-ID")
        if id == "" {
            return "", fmt.Errorf("no X-Account-ID")
        }
        return id, nil
    }

    func rateLimitResponse(w http.ResponseWriter, _ *http.Request) {
        w.Header().Set("Content-Type", "application/json")
        w.WriteHeader(http.StatusTooManyRequests)
        _ = json.NewEncoder(w).Encode(map[string]string{"error": "rate_limited"})
    }
    ```

    **Step B — Introduce Publisher interface in hub.go.** This makes NATS testable without spinning up a real server. Add at top of `services/gateway/internal/ws/hub.go` (after imports):

    ```go
    // Publisher abstracts NATS publishing so tests can inject a fake.
    // Real impl: *nats.Conn (which satisfies this interface via its Publish method).
    type Publisher interface {
        Publish(subject string, data []byte) error
    }

    // FrameHandler dispatches client-sent WebSocket frames to their destinations.
    // typing -> Publisher ("chat.typing.{convID}" fan-out); read -> HTTP POST to chat /internal/read.
    type FrameHandler interface {
        HandleFrame(ctx context.Context, accountID string, frame map[string]any) error
    }

    type natsFrameHandler struct {
        pub             Publisher
        chatInternalURL string
        internalSecret  string
        httpClient      *http.Client
        log             zerolog.Logger
        // Per-client typing throttle: 1 event per 500ms keyed on accountID.
        typingMu        sync.Mutex
        typingLast      map[string]time.Time
    }

    // NewFrameHandler constructs the default dispatcher wiring the given Publisher
    // for typing and HTTP for read-receipt delivery to the chat service.
    // In production pass a *nats.Conn as the Publisher.
    func NewFrameHandler(pub Publisher, chatInternalURL, internalSecret string, log zerolog.Logger) FrameHandler {
        return &natsFrameHandler{
            pub:             pub,
            chatInternalURL: chatInternalURL,
            internalSecret:  internalSecret,
            httpClient:      &http.Client{Timeout: 2 * time.Second},
            log:             log,
            typingLast:      make(map[string]time.Time),
        }
    }

    func (h *natsFrameHandler) HandleFrame(ctx context.Context, accountID string, frame map[string]any) error {
        t, _ := frame["type"].(string)
        payload, _ := frame["payload"].(map[string]any)
        switch t {
        case "typing":
            convID, _ := payload["conv_id"].(string)
            if convID == "" {
                return nil
            }
            // Throttle: 1 typing event per 500ms per account.
            h.typingMu.Lock()
            last := h.typingLast[accountID]
            if time.Since(last) < 500*time.Millisecond {
                h.typingMu.Unlock()
                return nil
            }
            h.typingLast[accountID] = time.Now()
            h.typingMu.Unlock()

            data, _ := json.Marshal(map[string]any{
                "account_id": accountID,
                "conv_id":    convID,
                "ts":         time.Now().Unix(),
            })
            if h.pub == nil {
                return nil // test mode — no publisher wired
            }
            return h.pub.Publish("chat.typing."+convID, data)

        case "read":
            convID, _ := payload["conv_id"].(string)
            msgID, _  := payload["msg_id"].(string)
            if convID == "" || msgID == "" {
                return nil
            }
            body, _ := json.Marshal(map[string]string{
                "account_id": accountID,
                "conv_id":    convID,
                "msg_id":     msgID,
            })
            req, err := http.NewRequestWithContext(ctx, http.MethodPost,
                h.chatInternalURL+"/internal/read", bytes.NewReader(body))
            if err != nil {
                return err
            }
            req.Header.Set("Content-Type", "application/json")
            req.Header.Set("X-Internal-Secret", h.internalSecret)
            resp, err := h.httpClient.Do(req)
            if err != nil {
                h.log.Warn().Err(err).Msg("read-receipt dispatch failed")
                return err
            }
            _, _ = io.Copy(io.Discard, resp.Body)
            resp.Body.Close()
            return nil

        default:
            return nil // ignore unknown types silently (forward-compat)
        }
    }
    ```

    **Step C — Replace `ServeClient`** signature to accept FrameHandler. Update `services/gateway/internal/ws/hub.go:95-117`:

    ```go
    // ServeClient runs the read+write loops for a single WebSocket client.
    // Blocks until the connection closes. Client-sent frames are dispatched
    // to the provided FrameHandler.
    func (h *Hub) ServeClient(ctx context.Context, c *Client, handler FrameHandler) {
        h.Register(c)
        defer h.Unregister(c)

        go func() {
            for env := range c.send {
                if err := wsjson.Write(ctx, c.conn, env); err != nil {
                    c.conn.Close(websocket.StatusInternalError, "write error")
                    return
                }
            }
        }()

        for {
            var frame map[string]any
            if err := wsjson.Read(ctx, c.conn, &frame); err != nil {
                break
            }
            if handler != nil {
                if err := handler.HandleFrame(ctx, c.AccountID, frame); err != nil {
                    h.log.Debug().Err(err).Str("account", c.AccountID).Msg("frame dispatch error")
                }
            }
        }
    }
    ```

    ```go
    // wsHandler upgrades HTTP connections to WebSocket and serves the client.
    // Auth uses Sec-WebSocket-Protocol per RFC 6455 — the client sends two
    // subprotocols: the marker "koto.auth.v1" and the raw JWT. The server
    // validates the JWT, echoes back ONLY the marker (never the JWT) via
    // AcceptOptions.Subprotocols.
    func wsHandler(hub *ws.Hub, tokenMgr *token.Manager, pub ws.Publisher, chatInternalAddr, internalSecret string, log zerolog.Logger) http.HandlerFunc {
        return func(w http.ResponseWriter, r *http.Request) {
            rawSub := r.Header.Get("Sec-WebSocket-Protocol")
            if rawSub == "" {
                http.Error(w, "missing subprotocol", http.StatusUnauthorized)
                return
            }
            var marker, jwtToken string
            for _, p := range strings.Split(rawSub, ",") {
                p = strings.TrimSpace(p)
                switch {
                case p == "koto.auth.v1":
                    marker = p
                case jwtToken == "":
                    jwtToken = p
                }
            }
            if marker == "" || jwtToken == "" {
                http.Error(w, "invalid subprotocol", http.StatusUnauthorized)
                return
            }

            claims, err := tokenMgr.Verify(jwtToken)
            if err != nil {
                http.Error(w, "invalid token", http.StatusUnauthorized)
                return
            }

            allowedOrigin := env("ALLOWED_ORIGIN", "")
            originPatterns := []string{"localhost:*"}
            if allowedOrigin != "" {
                originPatterns = []string{allowedOrigin, "localhost:*"}
            }
            conn, err := websocket.Accept(w, r, &websocket.AcceptOptions{
                Subprotocols:   []string{"koto.auth.v1"},
                OriginPatterns: originPatterns,
            })
            if err != nil {
                log.Error().Err(err).Msg("ws accept failed")
                return
            }

            client := ws.NewClient(claims.AccountID, conn)
            log.Debug().Str("account", claims.AccountID).Msg("ws client connected")
            handler := ws.NewFrameHandler(pub, chatInternalAddr, internalSecret, log)
            hub.ServeClient(r.Context(), client, handler)
            log.Debug().Str("account", claims.AccountID).Msg("ws client disconnected")
        }
    }
    ```
