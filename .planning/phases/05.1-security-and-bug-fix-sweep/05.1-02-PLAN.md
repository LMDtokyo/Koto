    **Step A — Create migration** `services/auth/migrations/003_prekey_sequence.sql`:
    ```sql
    -- Auth service — Phase 05.1 migration 003.
    -- Fixes TOCTOU race in PublishPreKeys by delegating id assignment to Postgres.
    -- Idempotent: existing rows keep their ids; the identity sequence starts at MAX(id)+1.
    -- Run with: psql -U $POSTGRES_USER -d koto -f 003_prekey_sequence.sql

    BEGIN;

    ALTER TABLE one_time_pre_keys
        ALTER COLUMN id ADD GENERATED BY DEFAULT AS IDENTITY;

    -- Seed the sequence past the highest existing id so inserts do not collide
    -- with pre-migration manually assigned values.
    SELECT setval(
        pg_get_serial_sequence('one_time_pre_keys', 'id'),
        COALESCE((SELECT MAX(id) FROM one_time_pre_keys), 0) + 1,
        false
    );

    COMMIT;
    ```

    **Step B — Rewrite** `PreKeyRepo.Save` in `services/auth/internal/infra/postgres/account_repo.go`. Find the existing `Save` method; replace its body so it uses `pgx.Batch` with `INSERT ... RETURNING id`. The signature stays `Save(ctx context.Context, keys []domain.OneTimePreKey) error` but the method writes the generated id back into each element of the slice via the slice index (pointer aliasing — iterate the slice by index not value).

    ```go
    // Save inserts one-time pre-keys and lets Postgres allocate ids via the
    // GENERATED BY DEFAULT AS IDENTITY sequence. The returned id is written
    // back into each element of keys (keys[i].ID).
    func (r *PreKeyRepo) Save(ctx context.Context, keys []domain.OneTimePreKey) error {
        if len(keys) == 0 {
            return nil
        }
        batch := &pgx.Batch{}
        for _, k := range keys {
            batch.Queue(
                `INSERT INTO one_time_pre_keys (account_id, key_data, used)
                 VALUES ($1, $2, false)
                 RETURNING id`,
                k.AccountID, k.KeyData,
            )
        }
        br := r.pool.SendBatch(ctx, batch)
        defer br.Close()

        for i := range keys {
            var id int32
            if err := br.QueryRow().Scan(&id); err != nil {
                return apperrors.Wrap(apperrors.ErrInternal, "save pre-keys", err)
            }
            keys[i].ID = uint32(id)
        }
        return nil
    }
    ```
    Verify the `pgx/v5` import path is already present; add if missing. Required import: `"github.com/jackc/pgx/v5"`.

    **Step C — Rewrite `PublishPreKeys`** in `services/auth/internal/app/service.go` lines 146-170. Replace with:
    ```go
    // PublishPreKeys stores a batch of fresh one-time pre-keys for an account.
    // Ids are assigned by Postgres (GENERATED BY DEFAULT AS IDENTITY) — the
    // repository writes them back into the input slice after insert.
    func (s *Service) PublishPreKeys(ctx context.Context, accountID string, keys [][]byte) (int, error) {
        if _, err := s.accounts.Get(ctx, accountID); err != nil {
            return 0, err
        }

        otks := make([]domain.OneTimePreKey, len(keys))
        for i, k := range keys {
            otks[i] = domain.OneTimePreKey{AccountID: accountID, KeyData: k}
            // ID intentionally unset — DB assigns via RETURNING.
        }
        if err := s.preKeys.Save(ctx, otks); err != nil {
            return 0, err
        }

        total, err := s.preKeys.Count(ctx, accountID)
        if err != nil {
            return 0, err
        }
        return total, nil
    }
    ```
    Note: `Count()` is still called — but AFTER insert and only to return the new grand total to the client. It is no longer used to pick ids.

    // Package app — Phase 05.1 tests.
    package app

    import (
        "context"
        "crypto/rand"
        "os"
        "sync"
        "testing"

        "github.com/jackc/pgx/v5/pgxpool"
        "github.com/koto-messenger/koto/services/auth/internal/domain"
        "github.com/koto-messenger/koto/services/auth/internal/infra/postgres"

    // TestPublishPreKeysConcurrent — SC-9. 10 goroutines × 20 keys → 200 distinct ids.
    func TestPublishPreKeysConcurrent(t *testing.T) {
        dsn := os.Getenv("TEST_POSTGRES_DSN")
        if dsn == "" {
            t.Skip("set TEST_POSTGRES_DSN to run (expects migrations 001 + 003 applied against koto DB)")
        }

        ctx := context.Background()
        pool, err := pgxpool.New(ctx, dsn)
        if err != nil { t.Fatalf("pgx pool: %v", err) }
        defer pool.Close()

        accountID := "test-concurrent-" + randomHex(8)
        // Seed a dummy account so the FK holds.
        _, _ = pool.Exec(ctx,
            `INSERT INTO accounts (id, identity_key, signed_pre_key, signed_pre_key_sig, signed_pre_key_id)
             VALUES ($1, $2, $2, $3, 1) ON CONFLICT DO NOTHING`,
            accountID, make([]byte, 32), make([]byte, 64))
        defer pool.Exec(ctx, `DELETE FROM accounts WHERE id=$1`, accountID)

        repo := postgres.NewPreKeyRepo(pool)      // confirm constructor name matches account_repo.go
        accRepo := postgres.NewAccountRepo(pool)   // confirm constructor name
        svc := New(accRepo, repo, nil, nil, 0)

        const goroutines = 10
        const perGoroutine = 20
        var wg sync.WaitGroup
        errCh := make(chan error, goroutines)

        for i := 0; i < goroutines; i++ {
            wg.Add(1)
            go func() {
                defer wg.Done()
                keys := make([][]byte, perGoroutine)
                for j := range keys {
                    b := make([]byte, 32)
                    _, _ = rand.Read(b)
                    keys[j] = b
                }
                if _, err := svc.PublishPreKeys(ctx, accountID, keys); err != nil {
                    errCh <- err
                }
            }()
        }
        wg.Wait()
        close(errCh)
        for e := range errCh {
            t.Fatalf("publish error: %v", e)
        }

        var count int
        if err := pool.QueryRow(ctx,
            `SELECT COUNT(DISTINCT id) FROM one_time_pre_keys WHERE account_id=$1`,
            accountID,
        ).Scan(&count); err != nil {
            t.Fatalf("count: %v", err)
        }
        want := goroutines * perGoroutine
        if count != want {
            t.Fatalf("distinct id count = %d, want %d (duplicates present → TOCTOU still unfixed)", count, want)
        }
    }

    func randomHex(n int) string {
        b := make([]byte, n)
        _, _ = rand.Read(b)
        const hexdigits = "0123456789abcdef"
        out := make([]byte, 2*n)
        for i, x := range b {
            out[2*i]   = hexdigits[x>>4]
            out[2*i+1] = hexdigits[x&0x0f]
        }
        return string(out)
    }

    // Ensure domain import stays referenced even if service gets refactored.
    var _ = domain.OneTimePreKey{}
