    **Step B — Add ScyllaDB implementation** to `services/chat/internal/infra/scylla/message_repo.go`. Append a new method to `ConversationRepo` (after the existing `Get` method at line 191):

    ```go
    // GetBatch returns all conversations whose ids appear in the input slice,
    // via a single CQL `WHERE id IN ?` query. Token-aware routing sends the
    // query to the coordinator that owns the first partition key.
    //
    // Performance: for 50 conversations this is one round-trip from the Go
    // service's perspective (compare vs 51 hops with the old per-id Get loop).
    // Under the hood Scylla fans out to each partition's owning node.
    func (r *ConversationRepo) GetBatch(ctx context.Context, ids []string) ([]domain.Conversation, error) {
        if len(ids) == 0 {
            return nil, nil
        }
        uuids := make([]gocql.UUID, 0, len(ids))
        for _, s := range ids {
            u, err := gocql.ParseUUID(s)
            if err != nil {
                // Skip invalid ids rather than fail the whole batch — matches existing
                // Get behaviour where a missing id returns ErrNotFound per-call.
                continue
            }
            uuids = append(uuids, u)
        }
        if len(uuids) == 0 {
            return nil, nil
        }

        iter := r.session.Query(
            `SELECT id, type, member_ids, created_at, last_msg_id, last_msg_at
             FROM conversations WHERE id IN ?`, uuids,
        ).WithContext(ctx).Iter()

        convs := make([]domain.Conversation, 0, len(uuids))
        for {
            var (
                c        domain.Conversation
                convType int8
            )
            if !iter.Scan(&c.ID, &convType, &c.MemberIDs, &c.CreatedAt, &c.LastMsgID, &c.LastMsgAt) {
                break
            }
            c.Type = domain.ConversationType(convType)
            convs = append(convs, c)
        }
        if err := iter.Close(); err != nil {
            return nil, apperrors.Wrap(apperrors.ErrInternal, "get conversations batch", err)
        }
        return convs, nil
    }
    ```
