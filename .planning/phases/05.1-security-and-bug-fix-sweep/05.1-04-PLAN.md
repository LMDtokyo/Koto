    ```go
    // Package http provides HTTP utilities for the bot service.
    // Exports SSRF-safe URL validation and an http.Client whose dialer
    // rechecks the resolved IP before connecting (DNS rebinding defence).
    package http

    import (
        "context"
        "errors"
        "fmt"
        "net"
        stdhttp "net/http"
        "net/url"
        "syscall"
        "time"
    )

    // ErrInvalidScheme is returned when the URL scheme is not https (or http in dev mode).
    var ErrInvalidScheme = errors.New("scheme must be https (or http in dev)")
    // ErrPrivateTarget is returned when the URL host resolves to a private / loopback / link-local address.
    var ErrPrivateTarget = errors.New("target resolves to a private, loopback, or link-local address")
    // ErrUnresolvable is returned when DNS lookup fails.
    var ErrUnresolvable = errors.New("unable to resolve hostname")

    // ValidateWebhookURL checks that rawURL points at a publicly reachable host.
    // When devMode is true the http scheme is accepted (for local testing against
    // httpbin etc). Every resolved IP address is checked against isBlockedIP; if
    // any resolves to a blocked range the call fails.
    func ValidateWebhookURL(ctx context.Context, rawURL string, devMode bool) error {
        u, err := url.Parse(rawURL)
        if err != nil {
            return fmt.Errorf("parse URL: %w", err)
        }
        if u.Scheme != "https" && !(devMode && u.Scheme == "http") {
            return ErrInvalidScheme
        }
        host := u.Hostname()
        if host == "" {
            return ErrInvalidScheme
        }

        ips, err := net.DefaultResolver.LookupIPAddr(ctx, host)
        if err != nil {
            return fmt.Errorf("%w: %v", ErrUnresolvable, err)
        }
        for _, ip := range ips {
            if isBlockedIP(ip.IP) {
                return fmt.Errorf("%w: %s → %s", ErrPrivateTarget, host, ip.IP)
            }
        }
        return nil
    }

    // isBlockedIP reports whether ip falls into a range that a webhook may not target.
    // Covers RFC1918 / loopback / link-local (including 169.254.169.254) / fc00::/7 /
    // unspecified / multicast / CGNAT (100.64.0.0/10).
    func isBlockedIP(ip net.IP) bool {
        if ip.IsLoopback() || ip.IsLinkLocalUnicast() ||
            ip.IsLinkLocalMulticast() || ip.IsMulticast() ||
            ip.IsPrivate() || ip.IsUnspecified() {
            return true
        }
        if v4 := ip.To4(); v4 != nil && v4[0] == 100 && v4[1]&0xC0 == 64 {
            return true
        }
        return false
    }

    // SafeClient returns an http.Client whose dialer rechecks the resolved IP
    // against isBlockedIP before connecting. Defends against DNS rebinding
    // between ValidateWebhookURL (registration time) and actual dispatch.
    func SafeClient(devMode bool, timeout time.Duration) *stdhttp.Client {
        dialer := &net.Dialer{
            Timeout: 5 * time.Second,
            Control: func(network, address string, _ syscall.RawConn) error {
                host, _, err := net.SplitHostPort(address)
                if err != nil {
                    return err
                }
                if ip := net.ParseIP(host); ip != nil && isBlockedIP(ip) {
                    return ErrPrivateTarget
                }
                return nil
            },
        }
        return &stdhttp.Client{
            Timeout: timeout,
            Transport: &stdhttp.Transport{
                DialContext: dialer.DialContext,
            },
        }
    }
    ```

    package http

    import (
        "context"
        "errors"
        "net"
        "strings"
        "testing"
    )

    func TestValidateWebhookURL(t *testing.T) {
        cases := []struct {
            name      string
            rawURL    string
            devMode   bool
            wantErr   bool
            wantErrIs error
        }{
            {"loopback",           "http://127.0.0.1/hook",              true,  true, ErrPrivateTarget},
            {"link-local-imds",    "http://169.254.169.254/",            true,  true, ErrPrivateTarget},
            {"rfc1918",            "http://10.0.0.1/hook",               true,  true, ErrPrivateTarget},
            {"unsupported-scheme", "ftp://example.com/hook",             true,  true, ErrInvalidScheme},
            {"localhost-string",   "http://localhost/hook",              true,  true, ErrPrivateTarget},
            {"http-not-in-dev",    "http://good-bot.example.com/hook",   false, true, ErrInvalidScheme},
            // The happy case may fail with ErrUnresolvable in sandboxed CI where DNS
            // is disabled; accept either nil OR an ErrUnresolvable wrapper.
            {"valid-public-https", "https://good-bot.example.com/hook",  false, false, nil},
        }
        for _, tc := range cases {
            t.Run(tc.name, func(t *testing.T) {
                err := ValidateWebhookURL(context.Background(), tc.rawURL, tc.devMode)
                if tc.wantErr && err == nil {
                    t.Fatalf("expected error, got nil")
                }
                if !tc.wantErr {
                    // Allow ErrUnresolvable in sandboxed environments (no DNS).
                    if err != nil && !errors.Is(err, ErrUnresolvable) {
                        t.Fatalf("unexpected error: %v", err)
                    }
                    return
                }
                if tc.wantErrIs != nil && !errors.Is(err, tc.wantErrIs) && !strings.Contains(err.Error(), tc.wantErrIs.Error()) {
                    t.Fatalf("want err %v, got %v", tc.wantErrIs, err)
                }
            })
        }
    }

    func TestIsBlockedIP(t *testing.T) {
        cases := []struct {
            ip      string
            blocked bool
        }{
            {"127.0.0.1",        true},
            {"169.254.169.254",  true},
            {"10.0.0.1",         true},
            {"172.16.0.1",       true},
            {"192.168.1.1",      true},
            {"100.64.0.1",       true},   // CGNAT
            {"::1",              true},
            {"fc00::1",          true},
            {"8.8.8.8",          false},
            {"1.1.1.1",          false},
            {"2606:4700:4700::1111", false},
        }
        for _, tc := range cases {
            t.Run(tc.ip, func(t *testing.T) {
                got := isBlockedIP(net.ParseIP(tc.ip))
                if got != tc.blocked {
                    t.Fatalf("isBlockedIP(%s) = %v, want %v", tc.ip, got, tc.blocked)
                }
            })
        }
    }

    ```go
    // Package nats — bot-service consumer. Subscribes to chat delivery events
    // and invokes each matching bot's webhook.
    package nats

    import (
        "bytes"
        "context"
        "encoding/json"
        stdhttp "net/http"
        "time"

        natsio "github.com/nats-io/nats.go"
        "github.com/jackc/pgx/v5/pgxpool"
        "github.com/rs/zerolog"

        bothttp "github.com/koto-messenger/koto/services/bot/internal/http"
    )

    // botDeliveryEvent mirrors chat.domain.DeliveryEvent — we deliberately duplicate
    // rather than import across service boundaries (same pattern as gateway).
    type botDeliveryEvent struct {
        MessageID      string    `json:"msg_id"`
        ConversationID string    `json:"conv_id"`
        SenderID       string    `json:"sender_id"`
        RecipientIDs   []string  `json:"recipient_ids"`
        Type           uint8     `json:"type"`
        Ciphertext     []byte    `json:"ciphertext"`
        SentAt         time.Time `json:"sent_at"`
    }

    // SubscribeDelivery starts a NATS subscription on chat.deliver.> and dispatches
    // every event whose recipient list contains a registered active bot account.
    // Sender-bot events are skipped to avoid webhook loops.
    func SubscribeDelivery(nc *natsio.Conn, pool *pgxpool.Pool, devMode bool, log zerolog.Logger) error {
        client := bothttp.SafeClient(devMode, 10*time.Second)
        _, err := nc.Subscribe("chat.deliver.>", func(msg *natsio.Msg) {
            var event botDeliveryEvent
            if err := json.Unmarshal(msg.Data, &event); err != nil {
                log.Error().Err(err).Msg("bot: bad delivery event")
                return
            }

            // Loop avoidance: if sender is itself a registered bot, skip.
            var senderIsBot bool
            if err := pool.QueryRow(context.Background(),
                `SELECT EXISTS (SELECT 1 FROM bots WHERE account_id = $1)`,
                event.SenderID,
            ).Scan(&senderIsBot); err != nil {
                log.Warn().Err(err).Msg("bot: sender check failed")
            }
            if senderIsBot {
                return
            }

            for _, rid := range event.RecipientIDs {
                var (
                    botID, webhookURL string
                )
                err := pool.QueryRow(context.Background(),
                    `SELECT id, webhook_url FROM bots WHERE account_id = $1 AND active = true`,
                    rid,
                ).Scan(&botID, &webhookURL)
                if err != nil || webhookURL == "" {
                    continue // not a registered bot recipient
                }

                payload := map[string]any{
                    "bot_id":         botID,
                    "conversation_id": event.ConversationID,
                    "sender_id":      event.SenderID,
                    "message_id":     event.MessageID,
                    "ciphertext":     event.Ciphertext,
                    "sent_at":        event.SentAt.Unix(),
                }
                body, _ := json.Marshal(payload)

                ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
                req, err := stdhttp.NewRequestWithContext(ctx, stdhttp.MethodPost, webhookURL, bytes.NewReader(body))
                if err != nil {
                    cancel()
                    log.Warn().Err(err).Str("bot", botID).Msg("bot: build webhook request failed")
                    continue
                }
                req.Header.Set("Content-Type", "application/json")

                resp, err := client.Do(req)
                cancel()
                if err != nil {
                    log.Warn().Err(err).Str("bot", botID).Str("webhook", webhookURL).Msg("bot: webhook delivery failed")
                    continue
                }
                resp.Body.Close()
            }
        })
        if err != nil {
            return err
        }
        log.Info().Str("subject", "chat.deliver.>").Msg("bot: NATS delivery consumer ready")
        return nil
    }
    ```
