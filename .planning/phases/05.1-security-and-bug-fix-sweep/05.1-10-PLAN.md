    /** Archive (or unarchive) a conversation. Archived rows are hidden from observeAll. */
    suspend fun setArchived(convId: String, archived: Boolean): Result<Unit> = runCatching {
        conversationDao.setArchived(convId, archived)
        if (archived) {
            // Archived rows should never also be pinned — avoids phantoms in the pinned strip.
            conversationDao.setPinned(convId, false)
        }
    }

    /** Pin or unpin a conversation to the top of the list. */
    suspend fun setPinned(convId: String, pinned: Boolean): Result<Unit> = runCatching {
        conversationDao.setPinned(convId, pinned)
    }

    /** Mute or unmute notifications for a conversation. */
    suspend fun setMuted(convId: String, muted: Boolean): Result<Unit> = runCatching {
        conversationDao.setMuted(convId, muted)
    }

    ```kotlin
    fun onArchive(convId: String) {
        viewModelScope.launch {
            chatRepository.setArchived(convId, true)
        }
    }

    fun onPin(convId: String) {
        viewModelScope.launch {
            val currently = state.value.conversations.any { it.id == convId && it.isPinned }
                         || state.value.pinnedConvs.any { it.id == convId }
            chatRepository.setPinned(convId, !currently)
        }
    }

    fun onMute(convId: String) {
        viewModelScope.launch {
            val currently = (state.value.conversations + state.value.pinnedConvs)
                .any { it.id == convId && it.isMuted }
            chatRepository.setMuted(convId, !currently)
        }
    }
    ```
