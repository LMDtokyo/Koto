**1. Replace tween screen transitions with spring (NAV-05):**

Remove the existing `tween()`-based `pushEnter`, `pushExit`, `popEnter`, `popExit` constants. Replace with spring-based specs. The ROADMAP Phase 4 success criterion specifies: dampingRatio=0.85, stiffness=380.

```kotlin
// Spring spec for screen-level transitions (NAV-05 requirement)
private val navSpringSpec = spring<IntOffset>(
    dampingRatio = 0.85f,
    stiffness    = 380f,
)
private val navFloatSpec  = spring<Float>(
    dampingRatio = 0.85f,
    stiffness    = 380f,
)

// Forward: new screen slides in from right 30% of width + fade
private val pushEnter = slideInHorizontally(
    animationSpec = navSpringSpec,
    initialOffsetX = { fullWidth -> (fullWidth * 0.30f).toInt() },
) + fadeIn(animationSpec = navFloatSpec)

// Current screen exits to left
private val pushExit = slideOutHorizontally(
    animationSpec = navSpringSpec,
    targetOffsetX = { fullWidth -> -(fullWidth * 0.15f).toInt() },
) + fadeOut(animationSpec = navFloatSpec)

// Previous screen returns from left
private val popEnter = slideInHorizontally(
    animationSpec = navSpringSpec,
    initialOffsetX = { fullWidth -> -(fullWidth * 0.15f).toInt() },
) + fadeIn(animationSpec = navFloatSpec)

// Current screen exits to right
private val popExit = slideOutHorizontally(
    animationSpec = navSpringSpec,
    targetOffsetX = { fullWidth -> (fullWidth * 0.30f).toInt() },
) + fadeOut(animationSpec = navFloatSpec)

// Simple spring crossfade for one-way transitions (onboarding â†’ conversations)
private val fadeInOnly  = fadeIn(animationSpec  = navFloatSpec)
private val fadeOutOnly = fadeOut(animationSpec = navFloatSpec)
```

**2. Add SharedTransitionLayout wrapping NavHost:**

Update `KotoNavGraph` signature to remain `@Composable fun KotoNavGraph(...)`. Wrap the existing NavHost in `SharedTransitionLayout { ... }`. Inside the SharedTransitionLayout lambda, `this` is the `SharedTransitionScope`. Pass it down to ConversationsScreen and ChatScreen:

```kotlin
@Composable
fun KotoNavGraph(
    navController    : NavHostController = rememberNavController(),
    startDestination : String = Screen.Onboarding.route,
) {
    val currentBackStackEntry by navController.currentBackStackEntryAsState()
    val currentRoute = currentBackStackEntry?.destination?.route

    // Determine whether to show bottom nav
    // Show only on root-level tabs; hide on chat screen, onboarding
    val showBottomNav = currentRoute in listOf(
        Screen.Conversations.route,
        "contacts",
        "calls",
        Screen.Settings.route,
    )

    SharedTransitionLayout {
        Scaffold(
            bottomBar = {
                AnimatedVisibility(
                    visible = showBottomNav,
                    enter   = slideInVertically { it } + fadeIn(),
                    exit    = slideOutVertically { it } + fadeOut(),
                ) {
                    KotoBottomNavBar(
                        currentRoute  = currentRoute,
                        onTabSelected = { dest ->
                            navController.navigate(dest.route) {
                                popUpTo(navController.graph.startDestinationId) { saveState = true }
                                launchSingleTop = true
                                restoreState    = true
                            }
                        }
                    )
                }
            },
            containerColor = Color.Transparent,  // Screens own their background
        ) { innerPadding ->
            NavHost(
                navController    = navController,
                startDestination = startDestination,
                modifier         = Modifier.padding(innerPadding),
                enterTransition     = { pushEnter  },
                exitTransition      = { pushExit   },
                popEnterTransition  = { popEnter   },
                popExitTransition   = { popExit    },
            ) {
                // Onboarding â€” one-way entry
                composable(
                    route           = Screen.Onboarding.route,
                    enterTransition = { fadeInOnly  },
                    exitTransition  = { fadeOutOnly },
                ) {
                    OnboardingScreen(
                        onRegistered = {
                            navController.navigate(Screen.Conversations.route) {
                                popUpTo(Screen.Onboarding.route) { inclusive = true }
                            }
                        }
                    )
                }

                // Conversations â€” passes SharedTransitionScope + AnimatedVisibilityScope
                composable(Screen.Conversations.route) {
                    ConversationsScreen(
                        onOpenChat = { convId ->
                            navController.navigate(Screen.Chat.build(convId))
                        },
                        onOpenSettings  = { navController.navigate(Screen.Settings.route) },
                        sharedTransitionScope    = this@SharedTransitionLayout,
                        animatedVisibilityScope  = this@composable,
                    )
                }

                // Chat â€” shared element scope propagation
                composable(
                    route     = Screen.Chat.route,
                    arguments = listOf(navArgument("convId") { type = NavType.StringType }),
                ) { back ->
                    val convId = back.arguments?.getString("convId") ?: return@composable
                    ChatScreen(
                        convId = convId,
                        onBack = { navController.popBackStack() },
                        sharedTransitionScope   = this@SharedTransitionLayout,
                        animatedVisibilityScope = this@composable,
                    )
                }

                // Settings
                composable(Screen.Settings.route) {
                    SettingsScreen(onBack = { navController.popBackStack() })
                }

                // Stub destinations for bottom nav tabs (Contacts, Calls)
                composable("contacts") {
                    PlaceholderScreen("Contacts")
                }
                composable("calls") {
                    PlaceholderScreen("Calls")
                }
            }
        }
    }
}

@Composable
private fun PlaceholderScreen(label: String) {
    Box(
        modifier         = Modifier.fillMaxSize().background(KotoTheme.colors.background),
        contentAlignment = Alignment.Center,
    ) {
        Text(label, style = KotoTheme.typography.titleLarge, color = KotoTheme.colors.onSurface)
    }
}
```

@Composable
private fun ChatTopBar(
    convId                  : String,
    displayName             : String,
    online                  : Boolean,
    onBack                  : () -> Unit,
    sharedTransitionScope   : SharedTransitionScope? = null,
    animatedVisibilityScope : AnimatedVisibilityScope? = null,
) {
    val colors = KotoTheme.colors

    val avatarMod = if (sharedTransitionScope != null && animatedVisibilityScope != null) {
        with(sharedTransitionScope) {
            Modifier.sharedElement(
                state = rememberSharedContentState(key = "avatar-$convId"),
                animatedVisibilityScope = animatedVisibilityScope,
                boundsTransform = { _, _ ->
                    spring(dampingRatio = 0.85f, stiffness = 380f)
                }
            )
        }
    } else {
        Modifier
    }

    TopAppBar(
        title = {
            Row(verticalAlignment = Alignment.CenterVertically) {
                // Avatar with shared element modifier
                Box(
                    modifier = avatarMod
                        .size(38.dp)
                        .clip(CircleShape)
                        .background(colors.primary.copy(alpha = 0.2f)),
                    contentAlignment = Alignment.Center,
                ) {
                    Text(
                        text  = displayName.firstOrNull()?.uppercase() ?: "?",
                        style = KotoTheme.typography.titleSmall,
                        color = colors.primary,
                    )
                }
                Spacer(Modifier.width(KotoTheme.spacing.md))

                    Text(displayName, style = KotoTheme.typography.titleMedium, color = colors.onSurface)
                    if (online) Text("Online", style = KotoTheme.typography.labelSmall, color = colors.online)
                }
            }
        },
        navigationIcon = {
            IconButton(onClick = onBack) {
                Icon(Icons.AutoMirrored.Filled.ArrowBack, "Back", tint = colors.onSurface)
            }
        },
        colors = TopAppBarColors(containerColor = colors.background, /* rest transparent */),
    )
}
